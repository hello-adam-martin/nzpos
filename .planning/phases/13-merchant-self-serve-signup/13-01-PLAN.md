    Create supabase/migrations/017_provision_store_rpc.sql:
    - CREATE OR REPLACE FUNCTION provision_store(p_auth_user_id UUID, p_store_name TEXT, p_slug TEXT, p_owner_email TEXT) RETURNS JSONB
    - LANGUAGE plpgsql SECURITY DEFINER (mirrors complete_pos_sale pattern from 005)
    - Body:
      1. INSERT INTO public.stores (name, slug, owner_auth_id, is_active) VALUES (p_store_name, p_slug, p_auth_user_id, true) RETURNING id INTO v_store_id
      2. INSERT INTO public.staff (store_id, auth_user_id, name, role) VALUES (v_store_id, p_auth_user_id, p_owner_email, 'owner')
      3. INSERT INTO public.store_plans (store_id) VALUES (v_store_id)
      4. RETURN jsonb_build_object('store_id', v_store_id, 'slug', p_slug)
    - EXCEPTION block: WHEN unique_violation THEN RAISE EXCEPTION 'SLUG_TAKEN:%', p_slug; WHEN OTHERS THEN RAISE EXCEPTION 'PROVISION_FAILED:%', SQLERRM
    - GRANT EXECUTE ON FUNCTION provision_store TO service_role
    - REVOKE EXECUTE ON FUNCTION provision_store FROM authenticated, anon, public
