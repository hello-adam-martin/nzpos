    ```typescript
    async function handleCompleteSale(cashTenderedCents?: number, splitCashCents?: number) {
      setIsProcessing(true)
      setSaleError(null)
      const totals = calcCartTotals(cart.items)

      let paymentMethod = cart.paymentMethod ?? 'eftpos'
      let notes: string | undefined
      if (splitCashCents != null && splitCashCents > 0) {
        paymentMethod = 'split'
        notes = `Split: ${formatNZD(splitCashCents)} cash, ${formatNZD(totals.totalCents - splitCashCents)} EFTPOS`
      }

      const result = await completeSale({
        channel: 'pos',
        status: 'completed',
        items: cart.items.map(item => ({
          product_id: item.productId,
          product_name: item.productName,
          unit_price_cents: item.unitPriceCents,
          quantity: item.quantity,
          discount_cents: item.discountCents,
          line_total_cents: item.lineTotalCents,
          gst_cents: item.gstCents,
        })),
        subtotal_cents: totals.subtotalCents,
        gst_cents: totals.gstCents,
        total_cents: totals.totalCents,
        discount_cents: cart.items.reduce((s, i) => s + i.discountCents, 0),
        payment_method: paymentMethod,
        cash_tendered_cents: cashTenderedCents ?? splitCashCents,
        notes,
      })

      setIsProcessing(false)

      if ('error' in result) {
        if (result.error === 'out_of_stock') {
          setSaleError(`Out of stock — ${result.message || 'item was just sold'}. Void this sale.`)
        } else {
          setSaleError(typeof result.error === 'string' ? result.error : 'Sale could not be recorded.')
        }
        return
      }

      dispatch({ type: 'SALE_COMPLETE', orderId: result.orderId })
      router.refresh()  // refresh stock counts after sale (POS-08)
    }
    ```
