       ```sql
       -- Enable RLS on all tables
       ALTER TABLE public.stores ENABLE ROW LEVEL SECURITY;
       ALTER TABLE public.staff ENABLE ROW LEVEL SECURITY;
       ALTER TABLE public.categories ENABLE ROW LEVEL SECURITY;
       ALTER TABLE public.products ENABLE ROW LEVEL SECURITY;
       ALTER TABLE public.orders ENABLE ROW LEVEL SECURITY;
       ALTER TABLE public.order_items ENABLE ROW LEVEL SECURITY;
       ALTER TABLE public.promo_codes ENABLE ROW LEVEL SECURITY;
       ALTER TABLE public.stripe_events ENABLE ROW LEVEL SECURITY;
       ALTER TABLE public.cash_sessions ENABLE ROW LEVEL SECURITY;

       -- Stores: owner can see their own store
       CREATE POLICY "owner_isolation" ON public.stores
         FOR ALL USING (id = (auth.jwt() -> 'app_metadata' ->> 'store_id')::UUID);

       -- All other tables: store_id must match JWT claim
       CREATE POLICY "tenant_isolation" ON public.staff
         FOR ALL USING (store_id = (auth.jwt() -> 'app_metadata' ->> 'store_id')::UUID);

       CREATE POLICY "tenant_isolation" ON public.categories
         FOR ALL USING (store_id = (auth.jwt() -> 'app_metadata' ->> 'store_id')::UUID);

       CREATE POLICY "tenant_isolation" ON public.products
         FOR ALL USING (store_id = (auth.jwt() -> 'app_metadata' ->> 'store_id')::UUID);

       -- Products: also allow public read for online storefront (anon role)
       -- NOTE: FOR SELECT policies use USING only. WITH CHECK applies to writes only.
       CREATE POLICY "public_read_active" ON public.products
         FOR SELECT USING (is_active = true);

       CREATE POLICY "tenant_isolation" ON public.orders
         FOR ALL USING (store_id = (auth.jwt() -> 'app_metadata' ->> 'store_id')::UUID);

       CREATE POLICY "tenant_isolation" ON public.order_items
         FOR ALL USING (store_id = (auth.jwt() -> 'app_metadata' ->> 'store_id')::UUID);

       CREATE POLICY "tenant_isolation" ON public.promo_codes
         FOR ALL USING (store_id = (auth.jwt() -> 'app_metadata' ->> 'store_id')::UUID);

       -- Promo codes: public read for online validation
       -- NOTE: FOR SELECT policies use USING only. WITH CHECK applies to writes only.
       CREATE POLICY "public_read_active" ON public.promo_codes
         FOR SELECT USING (is_active = true AND (expires_at IS NULL OR expires_at > now()));

       CREATE POLICY "tenant_isolation" ON public.stripe_events
         FOR ALL USING (store_id = (auth.jwt() -> 'app_metadata' ->> 'store_id')::UUID);

       CREATE POLICY "tenant_isolation" ON public.cash_sessions
         FOR ALL USING (store_id = (auth.jwt() -> 'app_metadata' ->> 'store_id')::UUID);
       ```

       IMPORTANT: The `public_read_active` policies for products and promo_codes use USING-only (no WITH CHECK — WITH CHECK is invalid for SELECT policies). These supplement the tenant_isolation policy — authenticated users hit tenant_isolation, anonymous users hit public_read_active.

    2. Create `supabase/migrations/003_auth_hook.sql` — custom access token hook per D-06 and RESEARCH Pattern 3:
       ```sql
       CREATE OR REPLACE FUNCTION public.custom_access_token_hook(event JSONB)
       RETURNS JSONB LANGUAGE plpgsql AS $$
       DECLARE
         claims JSONB;
         user_store_id UUID;
         user_role TEXT;
       BEGIN
         claims := event -> 'claims';
         SELECT s.store_id, s.role INTO user_store_id, user_role
         FROM public.staff s
         WHERE s.auth_user_id = (event ->> 'user_id')::UUID;
         IF user_store_id IS NOT NULL THEN
           IF jsonb_typeof(claims -> 'app_metadata') IS NULL THEN
             claims := jsonb_set(claims, '{app_metadata}', '{}');
           END IF;
           claims := jsonb_set(claims, '{app_metadata,store_id}', to_jsonb(user_store_id::TEXT));
           claims := jsonb_set(claims, '{app_metadata,role}', to_jsonb(user_role));
         END IF;
         event := jsonb_set(event, '{claims}', claims);
         RETURN event;
       END;
       $$;

       GRANT EXECUTE ON FUNCTION public.custom_access_token_hook TO supabase_auth_admin;
       REVOKE EXECUTE ON FUNCTION public.custom_access_token_hook FROM authenticated, anon, public;

       -- Grant supabase_auth_admin SELECT on staff table for the hook
       GRANT SELECT ON public.staff TO supabase_auth_admin;
       ```

       import { createServerClient } from '@supabase/ssr'
       import { NextRequest, NextResponse } from 'next/server'
       import type { Database } from '@/types/database'

       export async function createSupabaseMiddlewareClient(request: NextRequest) {
         let supabaseResponse = NextResponse.next({ request })
         const supabase = createServerClient<Database>(
           process.env.NEXT_PUBLIC_SUPABASE_URL!,
           process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
           {
             cookies: {
               getAll: () => request.cookies.getAll(),
               setAll: (cookiesToSet) => {
                 cookiesToSet.forEach(({ name, value, options }) =>
                   request.cookies.set(name, value)
                 )
                 supabaseResponse = NextResponse.next({ request })
                 cookiesToSet.forEach(({ name, value, options }) =>
                   supabaseResponse.cookies.set(name, value, options)
                 )
               },
             },
           }
         )
         return { supabase, response: supabaseResponse }
       }
