```typescript
let verifiedLoyaltyDiscountCents = 0
let verifiedLoyaltyCustomerId: string | null = null
let verifiedLoyaltyPointsRedeemed = 0

if (parsed.data.loyalty_points_to_redeem && parsed.data.loyalty_points_to_redeem > 0) {
  // Get the authenticated customer's ID
  // (Use existing auth to get customer email, then look up customer_id)
  const { data: customer } = await supabase
    .from('customers')
    .select('id')
    .eq('store_id', storeId)
    .eq('email', userEmail)  // from session auth
    .maybeSingle()

  if (customer) {
    // Server-side balance check — NEVER trust client
    const { data: loyaltyRow } = await supabase
      .from('loyalty_points')
      .select('points_balance')
      .eq('store_id', storeId)
      .eq('customer_id', customer.id)
      .maybeSingle()

    // Get redeem rate
    const { data: settings } = await supabase
      .from('loyalty_settings')
      .select('redeem_rate_cents, is_active')
      .eq('store_id', storeId)
      .maybeSingle()

    if (loyaltyRow && settings && settings.is_active && settings.redeem_rate_cents
        && loyaltyRow.points_balance >= parsed.data.loyalty_points_to_redeem) {
      const discountCents = parsed.data.loyalty_points_to_redeem * settings.redeem_rate_cents
      verifiedLoyaltyDiscountCents = Math.min(discountCents, totalCents)
      verifiedLoyaltyCustomerId = customer.id
      verifiedLoyaltyPointsRedeemed = parsed.data.loyalty_points_to_redeem
    }
  }
}
```
