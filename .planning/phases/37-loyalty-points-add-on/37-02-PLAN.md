**1. Create `src/actions/loyalty/getLoyaltySettings.ts`:**

```typescript
'use server'
import 'server-only'
import { resolveStoreAuth } from '@/lib/auth'  // or whichever auth helper provides storeId
import { createClient } from '@/lib/supabase/server'

export async function getLoyaltySettings() {
  const { storeId } = await resolveStoreAuth()
  const supabase = await createClient()

  const { data, error } = await supabase
    .from('loyalty_settings')
    .select('earn_rate_cents, redeem_rate_cents, is_active')
    .eq('store_id', storeId)
    .maybeSingle()

  if (error) return { error: error.message }
  return { data: data ?? { earn_rate_cents: null, redeem_rate_cents: null, is_active: true } }
}
```

**Rate display convention (important):**

- `redeem_rate_cents = 1` means "1 point = $0.01, so 100 points = $1 discount" (display as "100 points per $1 discount" in the input)
- Store internally as cents for consistency with all other monetary values

Input conversion:
- Earn rate: user enters "1" (points per dollar) -> store as `earn_rate_cents = 100` (cents per point)
- Redeem rate: user enters "100" (points per $1 discount) -> store as `redeem_rate_cents = 1` (cents per point)
