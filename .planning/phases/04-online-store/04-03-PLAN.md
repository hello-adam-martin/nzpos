export async function generateMetadata({ params }: { params: Promise<{ slug: string }> }): Promise<Metadata> {
  const { slug } = await params
  // Fetch product by slug from Supabase
  const product = await getProductBySlug(slug)
  if (!product) return { title: 'Product Not Found' }
  return {
    title: `${product.name} | Shop`,
    description: product.name,  // Use product name as description for v1
    openGraph: {
      title: product.name,
      images: product.image_url ? [product.image_url] : [],

  }
}

```typescript
// At top of file or separate file
'use client'
function AddToCartSection({ product }: { product: ProductData }) {
  const { dispatch, state } = useCart()
  const inCart = state.items.find(i => i.productId === product.id)
  const isSoldOut = product.stock_quantity <= 0

  return (
    <button
      onClick={() => {
        dispatch({ type: 'ADD_ITEM', product: { id: product.id, name: product.name, priceCents: product.price_cents, imageUrl: product.image_url, slug: product.slug, stockQuantity: product.stock_quantity } })
        dispatch({ type: 'OPEN_DRAWER' })
      }}
      disabled={isSoldOut}
      aria-disabled={isSoldOut}
      className={`...amber CTA classes...${isSoldOut ? ' opacity-50 cursor-not-allowed' : ' active:scale-[0.97]'}`}
    >
      {isSoldOut ? 'Sold Out' : inCart ? 'Add Another' : 'Add to Cart'}
    </button>
  )
}
```
